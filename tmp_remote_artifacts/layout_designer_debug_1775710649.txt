    def load_book_data(self):
        """Загружает текст и фотографии из файлов"""
        if self.book_json_path and os.path.exists(self.book_json_path):
            with open(self.book_json_path, 'r', encoding='utf-8') as f:
                book_data = json.load(f)

                if 'text' in book_data:
                    self.data['text'] = book_data['text']
                if 'photos' in book_data:
                    self.data['photos'] = book_data['photos']
    
    def setup_fonts(self):
        """Регистрирует шрифты для ReportLab"""
        try:
            # Пытаемся зарегистрировать PT Serif и PT Sans
            font_paths = {
                'PTSerif-Regular': '/usr/share/fonts/truetype/paratype/PTSerif-Regular.ttf',
                'PTSerif-Bold': '/usr/share/fonts/truetype/paratype/PTSerif-Bold.ttf',
                'PTSerif-Italic': '/usr/share/fonts/truetype/paratype/PTSerif-Italic.ttf',
                'PTSans-Regular': '/usr/share/fonts/truetype/paratype/PTSans-Regular.ttf',
                'PTSans-Bold': '/usr/share/fonts/truetype/paratype/PTSans-Bold.ttf'
            }
            
            for font_name, font_path in font_paths.items():
                if os.path.exists(font_path):
                    pdfmetrics.registerFont(TTFont(font_name, font_path))
                    
            # Проверяем что шрифты зарегистрированы
            registered = pdfmetrics.getRegisteredFontNames()
            if 'PTSerif-Regular' not in registered:
                print("WARNING: PT Serif не найден, используем Times-Roman")
                
        except Exception as e:
            print(f"WARNING: Ошибка регистрации шрифтов: {e}")
